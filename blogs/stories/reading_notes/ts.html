    # 剩余参数

    讲一个不定数量的参数放到一个数组中,根据使用者传入的参数决定函数有几个参数

    //计算所有参数的和
function sum(initalNum: number, ...nums: number[]) {
  let total = initalNum
  for (const num of nums) {
    total += num
  }
  return total
}

console.log(sum(20, 30))        //50
console.log(sum(20, 30, 40))    //90
console.log(sum(20, 30, 40, 50)) //140
